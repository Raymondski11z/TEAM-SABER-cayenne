#!/bin/bash
N=0

passwd -l root
sed -i "s=/bin/bash=/bin/false=g" /etc/passwd
sed -i "s=/bin/sh=/bin/false=g" /etc/passwd
sed -i "s=root:x:0:0"root:/root:/bin/false=root:x:0:0:root:/root:/bin/bash=g" /etc/passwd
read -p "Change user information? yes or no " C;

while [ $C == "yes" ]
do
let N=N+1
read -p "Enter user name" U;
P=$(head -n 100 /dev/urandom | md5sum)
echo $U:$P | chpasswd
echo "User " $N "'s password is "$P >> /home/cyberops/passwords;
read -p "again? yes or no " C;
done
